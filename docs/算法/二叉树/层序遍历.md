:smiling_imp: 利用队列的先进先出很容易办到，遍历每一层的节点
```js
var levelOrder = function(root) {
    const res = [];
    if (!root) {
        return res;
    }
    const q = [];
    q.push(root);
    while (q.length !== 0) {
        const currentLevelSize = q.length;
        //每一层节点的数量
        for (let i = 1; i <= currentLevelSize; ++i) {
            const node = q.shift();
            //弹出队列
            res.push(node.val);
            // 将下一层的节点放入队列
            if (node.left) q.push(node.left);
            if (node.right) q.push(node.right);
        }
    }
        
    return res;
};


```
